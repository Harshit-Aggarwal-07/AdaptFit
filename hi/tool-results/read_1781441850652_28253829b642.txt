   122→   122→function getResponseCategory(message: string): string {
   123→   123→  const lower = message.toLowerCase();
   124→   124→  if (/form|posture|alignment|technique|correct/i.test(lower)) return 'form';

   126→   126→  if (/mood|anxious|relax|stress|sad|happy|calm|feeling/i.test(lower)) return 'mood';
   127→   127→  if (/nutrition|eat|food|meal|diet|protein|calorie/i.test(lower)) return 'nutrition';
   128→   128→  if (/rehab|recovery|injury|healing|therapy|pain/i.test(lower)) return 'rehabilitation';
   129→   129→  if (/motivat|inspire|encourage|keep going|give up/i.test(lower)) return 'motivation';
   130→   130→  return 'default';

   141→   141→function useTypingEffect(text: string, speed: number = 20) {
   142→   142→  const [displayedText, setDisplayedText] = useState('');
   143→   143→  const [isComplete, setIsComplete] = useState(false);
   144→   144→
   145→   145→  useEffect(() => {
   146→   146→    setDisplayedText('');
   147→   147→    setIsComplete(false);
   148→   148→    let index = 0;
   149→   149→
   150→   150→    const timer = setInterval(() => {
   151→   151→      if (index < text.length) {
   152→   152→        setDisplayedText(text.slice(0, index + 1));
   153→   153→        index++;
   154→   154→      } else {
   155→   155→        setIsComplete(true);
   156→   156→        clearInterval(timer);
   157→   157→      }
   158→   158→    }, speed);
   159→   159→
   160→   160→    return () => clearInterval(timer);
   161→   161→  }, [text, speed]);
   162→   162→
   163→   163→  return { displayedText, isComplete };
   164→   164→}
